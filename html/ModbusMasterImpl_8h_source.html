534  template <typename M>
535  Poco::UInt16 sendFrame(const M& message)
536  {
537  if (_logger.debug())
538  {
539  _logger.debug("Sending frame: functionCode=%02x, slaveOrUnit=%02x", static_cast<unsigned>(message.functionCode), static_cast<unsigned>(message.slaveOrUnitAddress));
540  }
541 
542  int maxTrans = _pPort->maxSimultaneousTransactions();
543  if (_pThread && _pendingMap.size() >= maxTrans)
544  {
545  throw Poco::ProtocolException("Maximum number of pending requests exceeded");
546  }
547 
548  message.transactionID = _nextTransactionID;
549  if (maxTrans > 1)
550  {
551  ++_nextTransactionID;
552  }
553 
554  _pPort->sendFrame(message);
555 
556  if (_pThread)
557  {
558  _pendingMap[message.transactionID] = Poco::Timestamp();
559  }
560 
561  return message.transactionID;
562  }

611  void processFrame()
612  {
613  Poco::UInt8 fc = _pPort->receiveFrame(_timeout);
614  if (fc == 0)
615  {
616  _logger.notice("Invalid or imcomplete frame received.");
617  this->badFrameReceived(this);
618  return;
619  }
620 
621  if (_logger.debug())
622  {
623  _logger.debug("Processing frame: %02x", static_cast<unsigned>(fc));
624  }
625 
626  if ((fc & MODBUS_EXCEPTION_MASK) == MODBUS_EXCEPTION_MASK)
627  {
628  ModbusExceptionMessage message;
629  _pPort->decodeFrame(message);
630  this->exceptionReceived(this, message);
631  }
632  else switch (fc & MODBUS_FUNCTION_CODE_MASK)
633  {
634  case MODBUS_READ_COILS:
635  {
636  ReadCoilsResponse response;
637  _pPort->decodeFrame(response);
638  removePending(response.transactionID);
639  this->readCoilsResponseReceived(this, response);
640  }
641  break;
642 
643  case MODBUS_READ_DISCRETE_INPUTS:
644  {
645  ReadDiscreteInputsResponse response;
646  _pPort->decodeFrame(response);
647  removePending(response.transactionID);
648  this->readDiscreteInputsResponseReceived(this, response);
649  }
650  break;
651 
652  case MODBUS_READ_HOLDING_REGISTERS:
653  {
654  ReadHoldingRegistersResponse response;
655  _pPort->decodeFrame(response);
656  removePending(response.transactionID);
657  _logger.debug("Firing readHoldingRegistersResponseReceived event...");
658  this->readHoldingRegistersResponseReceived(this, response);
659  }
660  break;
661 
662  case MODBUS_READ_INPUT_REGISTERS:
663  {
664  ReadInputRegistersResponse response;
665  _pPort->decodeFrame(response);
666  removePending(response.transactionID);
667  this->readInputRegistersResponseReceived(this, response);
668  }
669  break;
670 
671  case MODBUS_WRITE_SINGLE_COIL:
672  {
673  WriteSingleCoilResponse response;
674  _pPort->decodeFrame(response);
675  removePending(response.transactionID);
676  this->writeSingleCoilResponseReceived(this, response);
677  }
678  break;
679 
680  case MODBUS_WRITE_SINGLE_REGISTER:
681  {
682  WriteSingleRegisterResponse response;
683  _pPort->decodeFrame(response);
684  removePending(response.transactionID);
685  this->writeSingleRegisterResponseReceived(this, response);
686  }
687  break;
688 
689  case MODBUS_READ_EXCEPTION_STATUS:
690  {
691  ReadExceptionStatusResponse response;
692  _pPort->decodeFrame(response);
693  removePending(response.transactionID);
694  this->readExceptionStatusResponseReceived(this, response);
695  }
696  break;
697 
698  case MODBUS_WRITE_MULTIPLE_COILS:
699  {
700  WriteMultipleCoilsResponse response;
701  _pPort->decodeFrame(response);
702  removePending(response.transactionID);
703  this->writeMultipleCoilsResponseReceived(this, response);
704  }
705  break;
706 
707  case MODBUS_WRITE_MULTIPLE_REGISTERS:
708  {
709  WriteMultipleRegistersResponse response;
710  _pPort->decodeFrame(response);
711  removePending(response.transactionID);
712  this->writeMultipleRegistersResponseReceived(this, response);
713  }
714  break;
715 
716  case MODBUS_MASK_WRITE_REGISTER:
717  {
718  MaskWriteRegisterResponse response;
719  _pPort->decodeFrame(response);
720  removePending(response.transactionID);
721  this->maskWriteRegisterResponseReceived(this, response);
722  }
723  break;
724 
725  case MODBUS_READ_WRITE_MULTIPLE_REGISTERS:
726  {
727  ReadWriteMultipleRegistersResponse response;
728  _pPort->decodeFrame(response);
729  removePending(response.transactionID);
730  this->readWriteMultipleRegistersResponseReceived(this, response);
731  }
732  break;
733 
734  case MODBUS_READ_FIFO_QUEUE:
735  {
736  ReadFIFOQueueResponse response;
737  _pPort->decodeFrame(response);
738  removePending(response.transactionID);
739  this->readFIFOQueueResponseReceived(this, response);
740  }
741  break;
742 
743  default:
744  {
745  GenericMessage response;
746  _pPort->decodeFrame(response);
747  removePending(response.transactionID);
748  this->responseReceived(this, response);
749  }
750  break;
751  }
