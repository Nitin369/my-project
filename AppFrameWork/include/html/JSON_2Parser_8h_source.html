317  template <typename S>
318  bool parseChar(CharIntType nextChar, S& source)
322  {
323  CharIntType nextClass, nextState;
324  unsigned char ch = static_cast<unsigned char>(CharTraits::to_char_type(nextChar));
325 
326  // Determine the character's class.
327  if (!_allowNullByte && ch == 0) return false;
328  if (ch >= 0x80)
329  {
330  nextClass = C_ETC;
331  CharIntType count = utf8CheckFirst(nextChar);
332  if (!count)
333  {
334  throw Poco::JSON::JSONException("Bad character.");
335  }
336 
337  char buffer[4];
338  buffer[0] = nextChar;
339  for (int i = 1; i < count; ++i)
340  {
341  int c = 0;
342  if (!source.nextChar(c)) throw Poco::JSON::JSONException("Invalid UTF8 sequence found");
343  buffer[i] = c;
344  }
345 
346  if (!Poco::UTF8Encoding::isLegal((unsigned char*) buffer, count))
347  {
348  throw Poco::JSON::JSONException("No legal UTF8 found");
349  }
350 
351  for (int i = 0; i < count; ++i)
352  {
353  parseBufferPushBackChar(buffer[i]);
354  }
355  return true;
356  }
357  else
358  {
359  nextClass = _asciiClass[nextChar];
360  if (nextClass <= xx) return false;
361  }
362 
363  addCharToParseBuffer(nextChar, nextClass);
364 
365  // Get the next _state from the _state transition table.
366  nextState = _stateTransitionTable[_state][nextClass];
367  if (nextState >= 0)
368  {
369  _state = nextState;
370  }
371  else
372  {
373  // Or perform one of the actions.
374  switch (nextState)
375  {
376  // Unicode character
377  case UC:
378  if (!decodeUnicodeChar()) return false;
379  // check if we need to read a second UTF-16 char
380  if (_utf16HighSurrogate) _state = D1;
381  else _state = ST;
382  break;
383  // _escaped char
384  case EX:
385  _escaped = 1;
386  _state = EC;
387  break;
388  // integer detected by minus
389  case MX:
390  _type = JSON_T_INTEGER;
391  _state = MI;
392  break;
393  // integer detected by zero
394  case ZX:
395  _type = JSON_T_INTEGER;
396  _state = ZE;
397  break;
398  // integer detected by 1-9
399  case IX:
400  _type = JSON_T_INTEGER;
401  _state = IT;
402  break;
403  // floating point number detected by exponent
404  case DE:
405  assertNotStringNullBool();
406  _type = JSON_T_FLOAT;
407  _state = E1;
408  break;
409  // floating point number detected by fraction
410  case DF:
411  assertNotStringNullBool();
412  _type = JSON_T_FLOAT;
413  _state = FX;
414  break;
415  // string begin "
416  case SB:
417  clearBuffer();
418  poco_assert(_type == JSON_T_NONE);
419  _type = JSON_T_STRING;
420  _state = ST;
421  break;
422 
423  // n
424  case NU:
425  poco_assert(_type == JSON_T_NONE);
426  _type = JSON_T_NULL;
427  _state = N1;
428  break;
429  // f
430  case FA:
431  poco_assert(_type == JSON_T_NONE);
432  _type = JSON_T_FALSE;
433  _state = F1;
434  break;
435  // t
436  case TR:
437  poco_assert(_type == JSON_T_NONE);
438  _type = JSON_T_TRUE;
439  _state = T1;
440  break;
441 
442  // closing comment
443  case CE:
444  _comment = 0;
445  poco_assert(_parseBuffer.size() == 0);
446  poco_assert(_type == JSON_T_NONE);
447  _state = _beforeCommentState;
448  break;
449 
450  // opening comment
451  case CB:
452  if (!_allowComments) return false;
453  parseBufferPopBackChar();
454  parseBuffer();
455  poco_assert(_parseBuffer.size() == 0);
456  poco_assert(_type != JSON_T_STRING);
457  switch (_stack[_top])
458  {
459  case MODE_ARRAY:
460  case MODE_OBJECT:
461  switch (_state)
462  {
463  case VA:
464  case AR:
465  _beforeCommentState = _state;
466  break;
467  default:
468  _beforeCommentState = OK;
469  break;
470  }
471  break;
472  default:
473  _beforeCommentState = _state;
474  break;
475  }
476  _type = JSON_T_NONE;
477  _state = C1;
478  _comment = 1;
479  break;
480  // empty }
481  case -9:
482  {
483  clearBuffer();
484  if (_pHandler) _pHandler->endObject();
485 
486  if (!pop(MODE_KEY)) return false;
487  _state = OK;
488  break;
489  }
490  // }
491  case -8:
492  {
493  parseBufferPopBackChar();
494  parseBuffer();
495  if (_pHandler) _pHandler->endObject();
496  if (!pop(MODE_OBJECT)) return false;
497  _type = JSON_T_NONE;
498  _state = OK;
499  break;
500  }
501  // ]
502  case -7:
503  {
504  parseBufferPopBackChar();
505  parseBuffer();
506  if (_pHandler) _pHandler->endArray();
507  if (!pop(MODE_ARRAY)) return false;
508  _type = JSON_T_NONE;
509  _state = OK;
510  break;
511  }
512  // {
513  case -6:
514  {
515  parseBufferPopBackChar();
516  if (_pHandler) _pHandler->startObject();
517  if (!push(MODE_KEY)) return false;
518  poco_assert(_type == JSON_T_NONE);
519  _state = OB;
520  break;
521  }
522  // [
523  case -5:
524  {
525  parseBufferPopBackChar();
526  if (_pHandler) _pHandler->startArray();
527  if (!push(MODE_ARRAY)) return false;
528  poco_assert(_type == JSON_T_NONE);
529  _state = AR;
530  break;
531  }
532  // string end "
533  case -4:
534  parseBufferPopBackChar();
535  switch (_stack[_top])
536  {
537  case MODE_KEY:
538  {
539  poco_assert(_type == JSON_T_STRING);
540  _type = JSON_T_NONE;
541  _state = CO;
542 
543  if (_pHandler)
544  {
545  _pHandler->key(std::string(_parseBuffer.begin(), _parseBuffer.size()));
546  }
547  clearBuffer();
548  break;
549  }
550  case MODE_ARRAY:
551  case MODE_OBJECT:
552  poco_assert(_type == JSON_T_STRING);
553  parseBuffer();
554  _type = JSON_T_NONE;
555  _state = OK;
556  break;
557  default:
558  return false;
559  }
560  break;
561 
562  // ,
563  case -3:
564  {
565  parseBufferPopBackChar();
566  parseBuffer();
567  switch (_stack[_top])
568  {
569  case MODE_OBJECT:
570  //A comma causes a flip from object mode to key mode.
571  if (!pop(MODE_OBJECT) || !push(MODE_KEY)) return false;
572  poco_assert(_type != JSON_T_STRING);
573  _type = JSON_T_NONE;
574  _state = KE;
575  break;
576  case MODE_ARRAY:
577  poco_assert(_type != JSON_T_STRING);
578  _type = JSON_T_NONE;
579  _state = VA;
580  break;
581  default:
582  return false;
583  }
584  break;
585  }
586  // :
587  case -2:
588  // A colon causes a flip from key mode to object mode.
589  parseBufferPopBackChar();
590  if (!pop(MODE_KEY) || !push(MODE_OBJECT)) return false;
591  poco_assert(_type == JSON_T_NONE);
592  _state = VA;
593  break;
594  //Bad action.
595  default:
596  return false;
597  }
598  }
599  return true;
600  }
