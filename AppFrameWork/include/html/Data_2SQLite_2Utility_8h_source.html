123  static bool registerUpdateHandler(sqlite3* pDB, CBT callbackFn, T* pParam)
132  {
133  typedef std::pair<CBT, T*> CBPair;
134  typedef std::multimap<sqlite3*, CBPair> CBMap;
135  typedef typename CBMap::iterator CBMapIt;
136  typedef std::pair<CBMapIt, CBMapIt> CBMapItPair;
137 
138  static CBMap retMap;
139  T* pRet = reinterpret_cast<T*>(eventHookRegister(pDB, callbackFn, pParam));
140 
141  if (pRet == 0)
142  {
143  if (retMap.find(pDB) == retMap.end())
144  {
145  retMap.insert(std::make_pair(pDB, CBPair(callbackFn, pParam)));
146  return true;
147  }
148  }
149  else
150  {
151  CBMapItPair retMapRange = retMap.equal_range(pDB);
152  for (CBMapIt it = retMapRange.first; it != retMapRange.second; ++it)
153  {
154  poco_assert (it->second.first != 0);
155  if ((callbackFn == 0) && (*pRet == *it->second.second))
156  {
157  retMap.erase(it);
158  return true;
159  }
160 
161  if ((callbackFn == it->second.first) && (*pRet == *it->second.second))
162  {
163  it->second.second = pParam;
164  return true;
165  }
166  }
167  }
168 
169  return false;
170  }
