79  {

84  {
85  if (this->gptr() == 0)
86  return fail;
87 
88  if (way == std::ios_base::beg)
89  {
90  newoff = 0;
91  }
92  else if (way == std::ios_base::cur)
93  {
94  // cur is not valid if both in and out are specified (Condition 3)
95  if ((which & std::ios_base::out) != 0)
96  return fail;
97  newoff = this->gptr() - this->eback();
98  }
99  else if (way == std::ios_base::end)

101  newoff = this->egptr() - this->eback();
102  }
103  else
104  {

107 
108  if ((newoff + off) < 0 || (this->egptr() - this->eback()) < (newoff + off))
109  return fail;
110  this->setg(this->eback(), this->eback() + newoff + off, this->egptr());
111  }
112 
113  if ((which & std::ios_base::out) != 0)
114  {
115  if (this->pptr() == 0)
116  return fail;
117 
118  if (way == std::ios_base::beg)
