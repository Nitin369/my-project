340  template <class T>
341  void addInternalExtract(const MetaColumn& mc)
352  {
353  std::string storage;
354 
355  switch (_storage)
356  {
357  case STORAGE_DEQUE_IMPL:
358  storage = DEQUE; break;
359  case STORAGE_VECTOR_IMPL:
360  storage = VECTOR; break;
361  case STORAGE_LIST_IMPL:
362  storage = LIST; break;
363  case STORAGE_UNKNOWN_IMPL:
364  storage = AnyCast<std::string>(session().getProperty("storage"));
365  break;
366  }
367 
368  if (storage.empty()) storage = DEQUE;
369 
370  if (0 == icompare(DEQUE, storage))
371  {
372  if (!isBulkExtraction())
373  addExtract(createExtract<std::deque<T> >(mc));
374  else
375  addExtract(createBulkExtract<std::deque<T> >(mc));
376  }
377  else if (0 == icompare(VECTOR, storage))
378  {
379  if (!isBulkExtraction())
380  addExtract(createExtract<std::vector<T> >(mc));
381  else
382  addExtract(createBulkExtract<std::vector<T> >(mc));
383  }
384  else if (0 == icompare(LIST, storage))
385  {
386  if (!isBulkExtraction())
387  addExtract(createExtract<std::list<T> >(mc));
388  else
389  addExtract(createBulkExtract<std::list<T> >(mc));
390  }
391  }
