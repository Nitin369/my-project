55 class RotateAtTimeStrategy: public RotateStrategy
57 {
58 public:
59  RotateAtTimeStrategy(const std::string& rtime):
60  _day(-1),
61  _hour(-1),
62  _minute(0)
63  {
64  if (rtime.empty())
65  throw InvalidArgumentException("Rotation time must be specified.");
66 
67  if ((rtime.find(',') != rtime.npos) && (rtime.find(':') == rtime.npos))
68  throw InvalidArgumentException("Invalid rotation time specified.");
69 
70  StringTokenizer timestr(rtime, ",:", StringTokenizer::TOK_TRIM | StringTokenizer::TOK_IGNORE_EMPTY);
71  int index = 0;
72 
73  switch (timestr.count())
74  {
75  case 3: // day,hh:mm
76  {
77  std::string::const_iterator it = timestr[index].begin();
78  _day = DateTimeParser::parseDayOfWeek(it, timestr[index].end());
79  ++index;
80  }
81  case 2: // hh:mm
82  _hour = NumberParser::parse(timestr[index]);
83  ++index;
84  case 1: // mm
85  _minute = NumberParser::parse(timestr[index]);
86  break;
87  default:
88  throw InvalidArgumentException("Invalid rotation time specified.");
89  }
90  getNextRollover();
91  }
92 
93  ~RotateAtTimeStrategy()
94  {
95  }
96 
97  bool mustRotate(LogFile* /*pFile*/)
98  {
99  if (DT() >= _threshold)
100  {
101  getNextRollover();
102  return true;
103  }
104  return false;
105  }
106 
107 private:
108  void getNextRollover()
109  {
110  Timespan tsp(0, 0, 1, 0, 1000); // 0,00:01:00.001
111  do
112  {
113  _threshold += tsp;
114  }
115  while (!(_threshold.minute() == _minute &&
116  (-1 == _hour || _threshold.hour() == _hour) &&
117  (-1 == _day || _threshold.dayOfWeek() == _day)));
118  // round to :00.0 seconds
119  _threshold.assign(_threshold.year(), _threshold.month(), _threshold.day(), _threshold.hour(), _threshold.minute());
120  }
121
