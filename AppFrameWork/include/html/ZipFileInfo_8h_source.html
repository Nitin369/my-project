399 inline void ZipFileInfo::setCompressionLevel(ZipCommon::CompressionLevel cl)
400 {
401  // bit 1 and 2 indicate the level
402  Poco::UInt16 val = static_cast<Poco::UInt16>(cl);
403  val <<= 1;
404  Poco::UInt16 mask = 0xfff9;
405  _rawInfo[GENERAL_PURPOSE_POS] = ((_rawInfo[GENERAL_PURPOSE_POS] & mask) | val);
