33 template <class TObj, class TArgs, bool withSender = true>
34 class Delegate: public AbstractDelegate<TArgs>
35 {
36 public:
37  typedef void (TObj::*NotifyMethod)(const void*, TArgs&);
38 
39  Delegate(TObj* obj, NotifyMethod method):
40  _receiverObject(obj),
41  _receiverMethod(method)
42  {
43  }
44 
45  Delegate(const Delegate& delegate):
46  AbstractDelegate<TArgs>(delegate),
47  _receiverObject(delegate._receiverObject),
48  _receiverMethod(delegate._receiverMethod)
49  {
50  }
51 
52  ~Delegate()
53  {
54  }
55 
56  Delegate& operator = (const Delegate& delegate)
57  {
58  if (&delegate != this)
59  {
60  this->_receiverObject = delegate._receiverObject;
61  this->_receiverMethod = delegate._receiverMethod;
62  }
63  return *this;
64  }
65 
66  bool notify(const void* sender, TArgs& arguments)
67  {
68  Mutex::ScopedLock lock(_mutex);
69  if (_receiverObject)
70  {
71  (_receiverObject->*_receiverMethod)(sender, arguments);
72  return true;
73  }
74  else return false;
75  }
76 
77  bool equals(const AbstractDelegate<TArgs>& other) const
78  {
79  const Delegate* pOtherDelegate = reinterpret_cast<const Delegate*>(other.unwrap());
80  return pOtherDelegate && _receiverObject == pOtherDelegate->_receiverObject && _receiverMethod == pOtherDelegate->_receiverMethod;
81  }
82 
83  AbstractDelegate<TArgs>* clone() const
84  {
85  return new Delegate(*this);
86  }
87 
88  void disable()
89  {
90  Mutex::ScopedLock lock(_mutex);
91  _receiverObject = 0;
92  }
93 
94 protected:
95  TObj* _receiverObject;
96  NotifyMethod _receiverMethod;
97  Mutex _mutex;
98 
99 private:
100  Delegate();
101 };
