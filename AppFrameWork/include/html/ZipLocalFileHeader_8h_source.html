317 inline void ZipLocalFileHeader::setCompressionLevel(ZipCommon::CompressionLevel cl)
318 {
319  // bit 1 and 2 indicate the level
320  Poco::UInt16 val = static_cast<Poco::UInt16>(cl);
321  val <<= 1;
322  Poco::UInt16 mask = 0xfff9;
323  _rawHeader[GENERAL_PURPOSE_POS] = ((_rawHeader[GENERAL_PURPOSE_POS] & mask) | val);
324 }
